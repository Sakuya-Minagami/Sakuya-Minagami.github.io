# 给定升序数组构造平衡二叉树
**平衡二叉树的中序遍历是递增的**
```js
var sortedArrayToBST = function(nums) {
    function f(start,end){
        if(start > end){
            return null
        }
        const mid = Math.floor((end + start)/2);
        return new TreeNode(nums[mid],f(start,mid - 1),f(mid + 1,end))
    }
    return f(0,nums.length-1)
};
```
