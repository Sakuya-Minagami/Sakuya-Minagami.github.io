### 循环下标
```
<li v-for="(item, index) in navList" 
	:key="index" 
	v-text="item.name" 
	:class="{ 'active-color': index === currentIndex }" 
	@click="navigate(item.path, index)" ></li>
```
methods
```
methods: { 
	navigate(path, index) { 
	this.currentIndex = index; 
	this.$router.push(path); 
	}, 
}
```
